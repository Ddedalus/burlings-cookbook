\begin{recipe}
    [% 
        preparationtime = {\unit[10]{min}},
        portion = {\portion{4}},
        bakingtime = {\unit[25]{min}}
    ]
    {Roast cauliflower and chickpea}

    \introduction{%
        Roast vegetables, spices and pulses - the quickest lunch ever!
        This roast can be served alone but to make it more filling you can add some grains or baguette.
    }

    \ingredients[13]{%
        1 & Cauliflower \\
        1 can & Chickpea \\
        250 g & Cherry tomatoes \\
        Marinade: &  \\
        3 tbs. & Oil \\
        2 tbs. & Soy sauce \\
        1 tbs. & Vinegar \\
        1 tbs. & Honey \\
        1 tbs. & Smoked paprika \\
        1 ts. & Hot paprika \\
        1 ts. & Coriander \\
        1 ts. & Cumin }

    \preparation{%
        \step Divide cauliflower into small florets. 
        \step Prepare marinade - mix everything well with a fork.
        \step Add chickpea to florets and cover all in marinade. Add cherry tomatoes.
        \step \underline{Bake at \unit[200]{\textcelcius}}
        \underline{for 25 min}. Half way through, mix vegetables in the oven. 
        \vspace{2cm}
    }

    \suggestion{%
        Under no circumstances skip smoked paprika or exchange it for normal one ;)
    }

    \hint{%
        You can serve it with fresh herbs: coriander or parsley; and seeds - pumpkin or sunflower.
    }

\end{recipe}
